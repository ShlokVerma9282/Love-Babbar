struct Node
{
    int data;
    struct Node* next;
    
    Node(int x){
        data = x;
        next = NULL;
    }
};
void printList(Node* node) 
{ 
    while (node != NULL) { 
        cout << node->data <<" "; 
        node = node->next; 
    }  
    cout<<"\n";
}

class Solution{
public:
    Node* divide(int N, Node *head){
        // code here
        Node *d1 = new Node(0);
        Node *d2 = new Node(0);
        Node *p1 = d1;
        Node *p2 = d2;
        Node *curr =head;
        while(curr){
            if(curr->data%2==0)
            {
                p2->next=curr;
                p2=p2->next;
            }
            else{
                p1->next=curr;
                p1=p1->next;
            }
            curr=curr->next;
        }
        p2->next = d1->next;
        p1->next = NULL;
        return d2->next;
    }
};
